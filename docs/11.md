# 내 예측, 얼마나 정확한가? 모델 평가

> 개발자의 질문: "내가 만든 예측 모델이 정말 좋은 건지 어떻게 알 수 있나요?"
> 좋은 모델을 만드는 것만큼 중요한 것이 "얼마나 좋은지" 정확히 측정하는 것입니다.

---

## 왜 배우나요?

열심히 만든 모델의 정확도가 60%라면...
- 정말 좋은 건가요?
- 그냥 매일 "오른다"고 예측해도 50%는 맞는데?
- 틀렸을 때 얼마나 손해가 났나요?

**모델 평가**는 이런 질문들에 제대로 답하는 방법입니다.

---

## 1. 주식 예측 모델 만들기

```python
import pandas as pd
import numpy as np
from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
from sklearn.linear_model import LogisticRegression
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import (accuracy_score, roc_auc_score,
                              precision_score, recall_score, confusion_matrix)
import matplotlib.pyplot as plt
import seaborn as sns

np.random.seed(42)

# 삼성전자 주가 600일치
days = 600
prices = 60000 + np.cumsum(np.random.randn(days) * 500)
volume = np.random.randint(5000000, 20000000, days)

df = pd.DataFrame({'close': prices, 'volume': volume})
df['ret']       = df['close'].pct_change()
df['ret_5']     = df['close'].pct_change(5)
df['ma5']       = df['close'].rolling(5).mean()
df['ma20']      = df['close'].rolling(20).mean()
df['vol_ratio'] = df['volume'] / df['volume'].rolling(10).mean()
df['target']    = (df['close'].shift(-1) > df['close']).astype(int)
df = df.dropna()

features = ['ret', 'ret_5', 'ma5', 'ma20', 'vol_ratio']
X = df[features].values
y = df['target'].values

split = int(len(X) * 0.8)
X_train, X_test = X[:split], X[split:]
y_train, y_test = y[:split], y[split:]

scaler = StandardScaler()
X_train_sc = scaler.fit_transform(X_train)
X_test_sc  = scaler.transform(X_test)

# 3가지 모델 학습
models = {
    '로지스틱 회귀':    LogisticRegression(random_state=42),
    '랜덤 포레스트':   RandomForestClassifier(n_estimators=100, max_depth=5, random_state=42),
    '그래디언트 부스팅': GradientBoostingClassifier(n_estimators=100, max_depth=3, random_state=42),
}

for name, m in models.items():
    m.fit(X_train_sc, y_train)

print("모든 모델 학습 완료!")
```

## 4. 투자 시뮬레이션 — 실제로 돈을 벌었나?

```python
def simulate_trading(y_pred_prob, actual_returns, threshold=0.55):
    """예측 신호로 투자했을 때 수익 시뮬레이션"""
    portfolio = [1.0]  # 시작 자산 1배
    buyhold   = [1.0]  # 그냥 보유했을 때

    for i in range(len(y_pred_prob)):
        ret = actual_returns[i]

        # 상승 확률이 threshold 이상이면 매수
        if y_pred_prob[i] >= threshold:
            portfolio.append(portfolio[-1] * (1 + ret))
        else:
            portfolio.append(portfolio[-1])  # 현금 보유

        buyhold.append(buyhold[-1] * (1 + ret))

    return np.array(portfolio), np.array(buyhold)

# 실제 수익률 (테스트 기간)
test_rets = df['ret'].values[split:split+len(y_test)]
test_rets = test_rets[:len(y_test)]

plt.figure(figsize=(12, 5))
for name, res in eval_results.items():
    port, bh = simulate_trading(res['prob'], test_rets)
    final_return = (port[-1] - 1) * 100
    plt.plot(port, linewidth=1.5, label=f"{name} (+{final_return:.1f}%)")

port_bh, _ = simulate_trading(np.ones(len(y_test)) * 0.6, test_rets)
_, bh = simulate_trading(np.ones(len(y_test)) * 0.6, test_rets)

plt.xlabel('거래일')
plt.ylabel('자산 배율')
plt.title('예측 모델 투자 시뮬레이션\n(시작 자산 = 1.0)')
```

## 2. 모델별 학습 및 신호 생성

```python
# ─── 성과 측정 함수 ──────────────────────────────────────────────────────
def backtest(probs, test_returns, threshold=0.55):
    """
    probs:         상승 확률 배열
    test_returns:  실제 일별 수익률
    threshold:     매수 기준 확률
    반환:          누적 수익률 곡선, 성과 지표 dict
    """
    portfolio = [1.0]
    buyhold   = [1.0]
    trades = 0

    for p, r in zip(probs, test_returns):
        bh_new = buyhold[-1] * (1 + r)
        if p >= threshold:
            portfolio.append(portfolio[-1] * (1 + r))
            trades += 1
        else:
            portfolio.append(portfolio[-1])
        buyhold.append(bh_new)

    port_arr = np.array(portfolio)
    bh_arr   = np.array(buyhold)

    # 최대 낙폭(MDD)
    peak     = np.maximum.accumulate(port_arr)
    drawdown = (port_arr - peak) / peak
    mdd      = drawdown.min()

    # 연환산 Sharpe Ratio
    daily_rets = np.diff(port_arr) / port_arr[:-1]
    sharpe = (daily_rets.mean() / (daily_rets.std() + 1e-9)) * np.sqrt(252) if len(daily_rets) > 1 else 0

    total_ret = (port_arr[-1] - 1) * 100
    bh_ret    = (bh_arr[-1]   - 1) * 100
    win_rate  = (daily_rets > 0).mean()

    return port_arr, bh_arr, {
        'total_return': round(total_ret, 2),
        'bh_return':    round(bh_ret,    2),
        'mdd':          round(mdd * 100, 2),
        'sharpe':       round(sharpe, 3),
        'win_rate':     round(win_rate, 3),
        'trades':       trades,
        'accuracy':     accuracy_score(y_test, (probs >= threshold).astype(int)),
    }


# ─── 모델 1: 랜덤 포레스트 ──────────────────────────────────────────────
rf = RandomForestClassifier(n_estimators=200, max_depth=5, random_state=42, n_jobs=-1)
rf.fit(X_tr_sc, y_train)
rf_probs = rf.predict_proba(X_te_sc)[:, 1]
rf_port, rf_bh, rf_metrics = backtest(rf_probs, test_rets)

# ─── 모델 2: LSTM 스타일 MLP ────────────────────────────────────────────
# 슬라이딩 윈도우 (20일) 기반
SEQ_LEN = 20
feat_vals = X_base
X_lstm, y_lstm = [], []
for i in range(SEQ_LEN, len(feat_vals)):
    X_lstm.append(feat_vals[i - SEQ_LEN:i].flatten())
    y_lstm.append(y_base[i] if i < len(y_base) else 0)

X_lstm = np.array(X_lstm[:-1])
y_lstm = np.array(y_lstm[:-1])

split_lstm = int(len(X_lstm) * 0.8)
sc_lstm = StandardScaler()
lstm_clf = MLPClassifier(hidden_layer_sizes=(128, 64), activation='tanh',
                          max_iter=400, random_state=42, early_stopping=True)
lstm_clf.fit(sc_lstm.fit_transform(X_lstm[:split_lstm]), y_lstm[:split_lstm])
lstm_probs = lstm_clf.predict_proba(sc_lstm.transform(X_lstm[split_lstm:]))[:, 1]
# test_rets를 LSTM 길이에 맞게 잘라냄
lstm_test_rets = test_rets[SEQ_LEN - 1:SEQ_LEN - 1 + len(lstm_probs)]
lstm_port, lstm_bh, lstm_metrics = backtest(lstm_probs, lstm_test_rets)

# ─── 모델 3: Transformer 기반 ───────────────────────────────────────────
def softmax(x):
    e = np.exp(x - x.max(axis=-1, keepdims=True))
    return e / e.sum(axis=-1, keepdims=True)

def transformer_feature(seq):
    np.random.seed(42)
    d = 8
    W_e = np.random.randn(seq.shape[1], d) * 0.1
    emb = seq @ W_e
    Wq = np.random.randn(d, d) * 0.05
    Wk = np.random.randn(d, d) * 0.05
    Wv = np.random.randn(d, d) * 0.05
    Q, K, V = emb @ Wq, emb @ Wk, emb @ Wv
    w = softmax((Q @ K.T) / np.sqrt(d))
    ctx = w @ V
    return (emb + ctx)[-1]

X_trans, y_trans = [], []
for i in range(SEQ_LEN, len(feat_vals)):
    X_trans.append(transformer_feature(feat_vals[i-SEQ_LEN:i]))
    y_trans.append(y_base[i] if i < len(y_base) else 0)

X_trans = np.array(X_trans[:-1])
y_trans = np.array(y_trans[:-1])
sp_tr = int(len(X_trans) * 0.8)
sc_tr = StandardScaler()
trans_clf = MLPClassifier(hidden_layer_sizes=(64, 32), max_iter=400,
                           random_state=42, early_stopping=True)
trans_clf.fit(sc_tr.fit_transform(X_trans[:sp_tr]), y_trans[:sp_tr])
trans_probs = trans_clf.predict_proba(sc_tr.transform(X_trans[sp_tr:]))[:, 1]
trans_test_rets = test_rets[SEQ_LEN - 1:SEQ_LEN - 1 + len(trans_probs)]
trans_port, trans_bh, trans_metrics = backtest(trans_probs, trans_test_rets)
```

## 핵심 정리

| 지표 | 의미 | 좋은 값 |
|------|------|---------|
| 전략 수익률 | 모델 신호로 거래했을 때 총 수익 | 전량보유 초과 |
| MDD | 최악의 순간 최대 손실 | -20% 이하 |
| Sharpe Ratio | 위험 대비 수익의 효율성 | 1.0 이상 |
| 승률 | 매수 신호의 적중률 | 55% 이상 |

**배운 모델 비교:**
- 랜덤 포레스트: 빠르고 안정적 — 기준 모델
- LSTM 스타일: 시계열 흐름 학습 — 추세 포착 유리
- Transformer: Self-Attention으로 중요 날짜 탐색 — 복잡한 패턴
