## 1. 퀀트 모델 평가 지표 전체 지도

### 회귀 지표 (수익률 예측)

| 지표 | 수식 | 퀀트 의미 |
|------|------|---------|
| **IC** | $\text{corr}(\hat{y}, y)$ | 피어슨 정보계수. ≥0.03 실전 사용 가능 |
| **Rank IC** | $\text{spearmanr}(\hat{y}, y)$ | 이상치에 강인, 더 신뢰할 수 있음 |
| **ICIR** | $\text{IC.mean()} / \text{IC.std()}$ | 정보비율. ≥0.5 이면 안정적 |
| **MAE** | $\frac{1}{n}\sum|y_i - \hat{y}_i|$ | 평균 절대 오차 |
| **RMSE** | $\sqrt{\frac{1}{n}\sum(y_i-\hat{y}_i)^2}$ | 큰 오차에 민감 |

### 분류 지표 (방향 예측)

| 지표 | 수식 | 퀀트 의미 |
|------|------|---------|
| **AUC-ROC** | AUROC | 0.5=랜덤, 0.55+ 실전 기준 |
| **Precision@K** | 상위 K% 신호 중 실제 상승 비율 | 포지션 진입 정확도 |
| **Max Drawdown** | 전략 최대 낙폭 | 위험 관리 핵심 |
| **Sharpe Ratio** | $(R-R_f)/\sigma_R$ | 위험 조정 수익률 |

```python
import numpy as np
import pandas as pd
from scipy.stats import spearmanr
from sklearn.metrics import roc_auc_score

def compute_quant_metrics(y_true: np.ndarray, y_pred: np.ndarray,
                          is_classification: bool = False) -> dict:
    """퀀트 핵심 지표 일괄 계산"""
    metrics = {}

    if is_classification:
        metrics['auc']       = roc_auc_score(y_true, y_pred)
        pred_bin = (y_pred > 0.5).astype(int)
        metrics['accuracy']  = (pred_bin == y_true).mean()
        # Precision@top20%
        k = max(1, int(len(y_pred) * 0.2))
        top_k_idx = np.argsort(y_pred)[-k:]
        metrics['precision@20'] = y_true[top_k_idx].mean()
    else:
        ic, _   = spearmanr(y_true, y_pred)
        pearson = np.corrcoef(y_true, y_pred)[0, 1]
        metrics['rank_ic']  = ic
        metrics['ic']       = pearson
        metrics['mae']      = np.abs(y_true - y_pred).mean()
        metrics['rmse']     = np.sqrt(((y_true - y_pred)**2).mean())

    return metrics


def compute_icir(ic_series: np.ndarray) -> float:
    """IC 시계열로부터 ICIR 계산"""
    return ic_series.mean() / (ic_series.std() + 1e-8)


def rolling_ic(y_true: np.ndarray, y_pred: np.ndarray, window: int = 20) -> np.ndarray:
    """롤링 IC 계산 (모델 안정성 진단)"""
    ics = []
    for i in range(window, len(y_true)):
        ic, _ = spearmanr(y_true[i-window:i], y_pred[i-window:i])
        ics.append(ic)
    return np.array(ics)

# 예시
np.random.seed(42)
n = 500
y_true_reg = np.random.randn(n)
y_pred_reg = 0.1 * y_true_reg + np.random.randn(n) * 0.99

reg_metrics = compute_quant_metrics(y_true_reg, y_pred_reg, is_classification=False)
print("회귀 지표:", {k: round(v, 4) for k, v in reg_metrics.items()})
```

## 2. 전략 백테스트 지표

```python
def backtest_long_short(y_pred: np.ndarray, y_true_ret: np.ndarray,
                        top_pct: float = 0.2, bot_pct: float = 0.2,
                        transaction_cost: float = 0.001) -> dict:
    """
    Long-Short 팩터 전략 백테스트
    - 매 기간 y_pred 상위 top_pct 매수, 하위 bot_pct 공매도
    """
    n = len(y_pred)
    k_top = max(1, int(n * top_pct))
    k_bot = max(1, int(n * bot_pct))

    top_idx = np.argsort(y_pred)[-k_top:]
    bot_idx = np.argsort(y_pred)[:k_bot]

    long_ret  = y_true_ret[top_idx].mean()
    short_ret = y_true_ret[bot_idx].mean()
    ls_ret    = long_ret - short_ret - 2 * transaction_cost

    return {
        'long_return':  long_ret,
        'short_return': short_ret,
        'ls_return':    ls_ret,
        'long_n':       k_top,
        'short_n':      k_bot,
    }


def compute_portfolio_metrics(returns: np.ndarray, freq: int = 252) -> dict:
    """포트폴리오 성과 지표"""
    r = np.array(returns)
    ann_return = r.mean() * freq
    ann_vol    = r.std() * np.sqrt(freq)
    sharpe     = ann_return / (ann_vol + 1e-8)

    cum = np.cumprod(1 + r)
    rolling_max = np.maximum.accumulate(cum)
    drawdown = (cum - rolling_max) / rolling_max
    max_dd = drawdown.min()

    return {
        'annual_return': ann_return,
        'annual_vol':    ann_vol,
        'sharpe_ratio':  sharpe,
        'max_drawdown':  max_dd,
        'calmar_ratio':  ann_return / (-max_dd + 1e-8),
    }

# 시뮬레이션
daily_strategy_ret = np.random.randn(252) * 0.01 + 0.0005
perf = compute_portfolio_metrics(daily_strategy_ret)
print("\n포트폴리오 성과:")
for k, v in perf.items():
    print(f"  {k}: {v:.4f}")
```

```python
from sklearn.model_selection import TimeSeriesSplit
from sklearn.linear_model import LogisticRegression

ax.set_xlabel('샘플 인덱스')
```

### Walk-Forward vs 블록 교차 검증 비교

```python
def purged_kfold_cv(X, y, model, n_splits=5, embargo_pct=0.01):
    """
    Purged K-Fold: 훈련/검증 경계에서 embargo(금지 구간) 적용
    - 피처 중첩으로 인한 누설 방지 (예: 롤링 팩터)
    """
    n = len(X)
    fold_size = n // n_splits
    embargo_size = int(n * embargo_pct)
    scores = []

    sc = StandardScaler()

    for i in range(n_splits):
        val_start = i * fold_size
        val_end   = val_start + fold_size

        # embargo: 검증 구간 앞뒤 제외
        tr_idx = np.concatenate([
            np.arange(0, max(0, val_start - embargo_size)),
            np.arange(min(n, val_end + embargo_size), n),
        ])
        val_idx = np.arange(val_start, val_end)

        if len(tr_idx) == 0 or len(val_idx) == 0:
            continue

        X_tr_sc = sc.fit_transform(X[tr_idx])
        X_va_sc = sc.transform(X[val_idx])
        model.fit(X_tr_sc, y[tr_idx])
        prob = model.predict_proba(X_va_sc)[:, 1]
        if len(np.unique(y[val_idx])) > 1:
            scores.append(roc_auc_score(y[val_idx], prob))

    return np.array(scores)

scores_purged = purged_kfold_cv(X, y, LogisticRegression(C=1.0, max_iter=1000))
print(f"\nPurged K-Fold AUC: {scores_purged.mean():.4f} ± {scores_purged.std():.4f}")
```

## 4. 하이퍼파라미터 튜닝 방법론

### GridSearchCV vs RandomSearch vs Optuna

```python
import optuna
from lightgbm import LGBMClassifier
from sklearn.model_selection import TimeSeriesSplit

optuna.logging.set_verbosity(optuna.logging.WARNING)

def create_lgb_study(X_tr, y_tr, n_splits=5, n_trials=100):
    """Optuna TPE로 LightGBM 하이퍼파라미터 최적화"""
    tscv = TimeSeriesSplit(n_splits=n_splits)

    def objective(trial):
        params = {
            'n_estimators':      trial.suggest_int('n_estimators', 50, 500),
            'learning_rate':     trial.suggest_float('learning_rate', 0.005, 0.3, log=True),
            'num_leaves':        trial.suggest_int('num_leaves', 15, 127),
            'max_depth':         trial.suggest_int('max_depth', 3, 10),
            'min_child_samples': trial.suggest_int('min_child_samples', 10, 100),
            'subsample':         trial.suggest_float('subsample', 0.5, 1.0),
            'colsample_bytree':  trial.suggest_float('colsample_bytree', 0.5, 1.0),
            'reg_alpha':         trial.suggest_float('reg_alpha', 1e-4, 10.0, log=True),
            'reg_lambda':        trial.suggest_float('reg_lambda', 1e-4, 10.0, log=True),
            'random_state': 42, 'n_jobs': 1, 'verbose': -1,
        }
        model = LGBMClassifier(**params)
        fold_aucs = []
        for tr_idx, val_idx in tscv.split(X_tr):
            model.fit(X_tr[tr_idx], y_tr[tr_idx])
            prob = model.predict_proba(X_tr[val_idx])[:, 1]
            if len(np.unique(y_tr[val_idx])) > 1:
                fold_aucs.append(roc_auc_score(y_tr[val_idx], prob))

        return np.mean(fold_aucs) if fold_aucs else 0.5

    study = optuna.create_study(direction='maximize',
                                 sampler=optuna.samplers.TPESampler(seed=42),
                                 pruner=optuna.pruners.MedianPruner())
    study.optimize(objective, n_trials=n_trials, show_progress_bar=False)
    return study

# 실행
sc = StandardScaler()
X_sc = sc.fit_transform(X)
split = int(len(X) * 0.8)
study = create_lgb_study(X_sc[:split], y[:split], n_trials=50)

print(f"최적 AUC: {study.best_value:.4f}")
print(f"최적 파라미터: {study.best_params}")
```

## 5. 과적합 진단 및 방지

```python
def plot_learning_curves(model, X, y, train_sizes=None, cv=5):
    """학습 곡선으로 과/과소적합 진단"""
    from sklearn.model_selection import learning_curve
    if train_sizes is None:
        train_sizes = np.linspace(0.1, 1.0, 10)

    tscv = TimeSeriesSplit(n_splits=cv)
    tr_sizes, tr_scores, val_scores = learning_curve(
        model, X, y, train_sizes=train_sizes,
        cv=tscv, scoring='roc_auc', n_jobs=-1,
    )

    plt.figure(figsize=(8, 5))
    plt.plot(tr_sizes, tr_scores.mean(axis=1), 'b-o', label='훈련 AUC')
    plt.fill_between(tr_sizes,
                     tr_scores.mean(1) - tr_scores.std(1),
                     tr_scores.mean(1) + tr_scores.std(1), alpha=0.2, color='b')
    plt.plot(tr_sizes, val_scores.mean(axis=1), 'r-o', label='검증 AUC')
    plt.fill_between(tr_sizes,
                     val_scores.mean(1) - val_scores.std(1),
                     val_scores.mean(1) + val_scores.std(1), alpha=0.2, color='r')
    plt.xlabel('훈련 샘플 수')
    plt.ylabel('AUC')
    plt.title('학습 곡선 (과적합 진단)')
    plt.legend()
    plt.tight_layout()
    plt.savefig('learning_curve.png', dpi=120)
    print("저장: learning_curve.png")

    # 진단
    gap = tr_scores.mean(1)[-1] - val_scores.mean(1)[-1]
    print(f"\n훈련/검증 최종 AUC 격차: {gap:.4f}")
    if gap > 0.05:
        print("→ 과적합 의심: 정규화 강화, 모델 단순화, 데이터 증가 권장")
    elif val_scores.mean(1)[-1] < 0.55:
        print("→ 과소적합: 모델 복잡도 증가, 팩터 추가, 피처 엔지니어링 권장")
    else:
        print("→ 적절한 적합 상태")

from sklearn.ensemble import RandomForestClassifier
plot_learning_curves(
    RandomForestClassifier(n_estimators=100, max_depth=4, random_state=42),
    sc.fit_transform(X), y, cv=3,
)
```

## 핵심 요약

| 주제 | 핵심 포인트 |
|------|----------|
| IC/Rank IC | 퀀트 모델 평가의 핵심. IC ≥ 0.03, ICIR ≥ 0.5 |
| TimeSeriesSplit | 시계열 CV의 기본. `shuffle=False` 절대 준수 |
| Purged K-Fold | 롤링 팩터 사용 시 누설 방지를 위한 Embargo 필수 |
| Optuna TPE | GridSearch보다 효율적. 파라미터 중요도도 파악 |
| 학습 곡선 | 훈련/검증 격차 > 0.05 → 과적합, 검증 AUC < 0.55 → 과소적합 |
| 최종 평가 | 테스트 데이터는 **단 한 번만** 사용. 재평가 금지 |
