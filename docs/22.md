## 1. CNN 핵심 원리

CNN은 **지역적 패턴(local pattern)**을 계층적으로 학습합니다.

```
입력 시계열: [−0.3, 0.1, 0.5, 0.8, 0.6, 0.3, −0.1]
                    ↓ 커널 [−1, 0, 1] (크기 3) 슬라이딩
합성곱 출력: [0.4, 0.4, 0.3, −0.5, −0.4]
                    ↓ ReLU + MaxPooling(2)
풀링 출력:   [0.4, 0.3]
```

### 2D vs 1D CNN

| | 2D CNN | 1D CNN |
|--|--------|--------|
| 입력 | (H, W, C) 이미지 | (T, C) 시계열 |
| 커널 | (kH, kW) | (k,) |
| 퀀트 적용 | 캔들스틱 차트 이미지 | OHLCV 시계열 |

## 2. 1D CNN으로 시계열 패턴 분류

### 데이터 준비 (슬라이딩 윈도우)

```python
import numpy as np
import torch
import torch.nn as nn
from torch.utils.data import DataLoader, TensorDataset
from sklearn.metrics import roc_auc_score, classification_report
import matplotlib.pyplot as plt

np.random.seed(42)
torch.manual_seed(42)

# 시계열 생성 (상승/하락/횡보 3클래스)
def generate_price_patterns(n_samples: int = 3000, window: int = 20):
    """가격 패턴 시뮬레이션: 0=하락, 1=횡보, 2=상승"""
    X_list, y_list = [], []
    for _ in range(n_samples):
        label = np.random.randint(0, 3)
        noise = np.random.randn(window) * 0.3
        if label == 2:    # 상승
            trend = np.linspace(0, 1.5, window) + noise
        elif label == 0:  # 하락
            trend = np.linspace(0, -1.5, window) + noise
        else:             # 횡보
            trend = noise

        # 정규화
        trend = (trend - trend.mean()) / (trend.std() + 1e-8)
        X_list.append(trend)
        y_list.append(label)

    return np.array(X_list, dtype=np.float32), np.array(y_list)

WINDOW = 20
X_data, y_data = generate_price_patterns(n_samples=5000, window=WINDOW)
print(f"데이터 형태: {X_data.shape}, 클래스 분포: {np.bincount(y_data)}")

# (N, T) → (N, 1, T): 채널 차원 추가
X_data = X_data[:, np.newaxis, :]  # (5000, 1, 20)

split_tr  = int(len(X_data) * 0.7)
split_val = int(len(X_data) * 0.85)

def to_tensor(arr):
    return torch.tensor(arr)

X_tr  = to_tensor(X_data[:split_tr])
X_val = to_tensor(X_data[split_tr:split_val])
X_te  = to_tensor(X_data[split_val:])
y_tr  = torch.tensor(y_data[:split_tr])
y_val = torch.tensor(y_data[split_tr:split_val])
y_te  = torch.tensor(y_data[split_val:])

train_loader = DataLoader(TensorDataset(X_tr, y_tr), batch_size=64, shuffle=True)
val_loader   = DataLoader(TensorDataset(X_val, y_val), batch_size=256)
```

### 1D CNN 모델

```python
class CNN1D(nn.Module):
    """
    입력: (batch, channels, time_steps)
    출력: (batch, n_classes)
    """
    def __init__(self, in_channels: int = 1, n_classes: int = 3):
        super().__init__()
        self.conv_block1 = nn.Sequential(
            nn.Conv1d(in_channels, 32, kernel_size=3, padding=1),
            nn.BatchNorm1d(32),
            nn.ReLU(),
            nn.MaxPool1d(kernel_size=2),  # 시퀀스 길이 절반
        )
        self.conv_block2 = nn.Sequential(
            nn.Conv1d(32, 64, kernel_size=3, padding=1),
            nn.BatchNorm1d(64),
            nn.ReLU(),
            nn.MaxPool1d(kernel_size=2),
        )
        self.conv_block3 = nn.Sequential(
            nn.Conv1d(64, 128, kernel_size=3, padding=1),
            nn.BatchNorm1d(128),
            nn.ReLU(),
        )
        self.gap = nn.AdaptiveAvgPool1d(1)  # Global Average Pooling → 가변 길이 입력 처리
        self.classifier = nn.Sequential(
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(64, n_classes),
        )

    def forward(self, x):
        x = self.conv_block1(x)
        x = self.conv_block2(x)
        x = self.conv_block3(x)
        x = self.gap(x).squeeze(-1)  # (batch, 128)
        return self.classifier(x)

model = CNN1D(in_channels=1, n_classes=3)
print(model)
total_params = sum(p.numel() for p in model.parameters())
print(f"파라미터 수: {total_params:,}")

# 입력 흐름 확인
dummy = torch.randn(4, 1, WINDOW)
print(f"\n입력: {dummy.shape}")
with torch.no_grad():
    out = model(dummy)
print(f"출력: {out.shape}")  # (4, 3)
```

### 훈련

```python
criterion = nn.CrossEntropyLoss()
optimizer = torch.optim.Adam(model.parameters(), lr=1e-3)
scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=50)

def accuracy(logits, targets):
    return (logits.argmax(dim=1) == targets).float().mean().item()

train_accs, val_accs = [], []

for epoch in range(1, 51):
    model.train()
    tr_acc_sum, tr_count = 0, 0
    for X_b, y_b in train_loader:
        logits = model(X_b)
        loss = criterion(logits, y_b)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        tr_acc_sum += accuracy(logits, y_b) * len(X_b)
        tr_count += len(X_b)
    scheduler.step()

    model.eval()
    with torch.no_grad():
        val_logits = model(X_val)
        val_acc = accuracy(val_logits, y_val)

    train_accs.append(tr_acc_sum / tr_count)
    val_accs.append(val_acc)

    if epoch % 10 == 0:
        print(f"Epoch {epoch:2d} | Train Acc={train_accs[-1]:.4f} | Val Acc={val_accs[-1]:.4f}")

# 최종 테스트
model.eval()
with torch.no_grad():
    test_logits = model(X_te)
    test_preds  = test_logits.argmax(dim=1).numpy()
    test_probs  = torch.softmax(test_logits, dim=1).numpy()

print("\n분류 보고서:")
print(classification_report(y_te.numpy(), test_preds,
                             target_names=['하락', '횡보', '상승']))
```

## 3. OHLCV 다채널 CNN

실전에서는 종가(C) 하나가 아니라 **OHLCV 5채널** 또는 기술적 지표를 추가 채널로 사용합니다.

```python
class MultiChannelCNN(nn.Module):
    """OHLCV 5채널 입력 CNN"""
    def __init__(self, n_channels: int = 5, seq_len: int = 20, n_classes: int = 2):
        super().__init__()
        self.net = nn.Sequential(
            # 채널 간 공간적 관계 학습
            nn.Conv1d(n_channels, 64, kernel_size=3, padding=1),
            nn.BatchNorm1d(64), nn.ReLU(),
            nn.Conv1d(64, 128, kernel_size=3, padding=1),
            nn.BatchNorm1d(128), nn.ReLU(),
            nn.AdaptiveAvgPool1d(1),
        )
        self.head = nn.Linear(128, n_classes)

    def forward(self, x):
        # x: (batch, n_channels, seq_len)
        return self.head(self.net(x).squeeze(-1))

# 입력 형태: (batch, 5, 20) — OHLCV × 20일 윈도우
multi_model = MultiChannelCNN(n_channels=5, seq_len=20, n_classes=2)
dummy_ohlcv = torch.randn(8, 5, 20)
print(f"OHLCV 입력: {dummy_ohlcv.shape} → 출력: {multi_model(dummy_ohlcv).shape}")
```

## 4. Grad-CAM으로 CNN 해석

어떤 시점(time step)이 분류 결정에 중요한지 시각화합니다.

```python
def compute_gradcam_1d(model, x_input, target_class, target_layer_name='conv_block3'):
    """1D Grad-CAM: 입력 시계열에서 중요한 구간 파악"""
    model.eval()
    activations, gradients = {}, {}

    def forward_hook(module, input, output):
        activations['feat'] = output

    def backward_hook(module, grad_in, grad_out):
        gradients['feat'] = grad_out[0]

    # 대상 레이어에 훅 등록
    target_layer = dict(model.named_children())[target_layer_name]
    fwd_handle = target_layer.register_forward_hook(forward_hook)
    bwd_handle = target_layer.register_backward_hook(backward_hook)

    logits = model(x_input)
    model.zero_grad()
    logits[0, target_class].backward()

    # CAM 계산
    grads   = gradients['feat'][0]         # (C, T)
    acts    = activations['feat'][0]       # (C, T)
    weights = grads.mean(dim=1)            # (C,)
    cam     = (weights[:, None] * acts).sum(dim=0)  # (T,)
    cam     = torch.relu(cam)
    cam     = (cam - cam.min()) / (cam.max() - cam.min() + 1e-8)

    fwd_handle.remove()
    bwd_handle.remove()

    return cam.detach().numpy()

# 상승 패턴 샘플 하나 분석
sample_idx = np.where(y_te.numpy() == 2)[0][0]
sample_input = X_te[sample_idx:sample_idx+1]
cam = compute_gradcam_1d(model, sample_input, target_class=2)

plt.figure(figsize=(10, 4))
time_axis = np.arange(WINDOW)
price_seq = sample_input[0, 0].numpy()

ax1 = plt.gca()
ax1.plot(time_axis, price_seq, 'b-o', label='주가 패턴', markersize=4)
ax1.set_ylabel('정규화 가격')
ax2 = ax1.twinx()
ax2.bar(time_axis, cam, alpha=0.4, color='red', label='Grad-CAM 중요도')
ax2.set_ylabel('중요도')
plt.title('Grad-CAM: 상승 예측에서 중요한 시점')
ax1.legend(loc='upper left')
ax2.legend(loc='upper right')
plt.tight_layout()
plt.savefig('gradcam_1d.png', dpi=120)
print("저장: gradcam_1d.png")
```

## 핵심 요약

- **Conv1d**: 시계열에서 지역적 패턴 (모멘텀, 반전) 자동 학습
- **Global Average Pooling**: 가변 길이 시계열을 고정 크기 벡터로 변환
- **다채널 입력**: OHLCV, 기술적 지표 등 여러 시계열을 채널로 쌓기
- **Grad-CAM**: CNN이 "어떤 시점"에 주목했는지 해석 가능
