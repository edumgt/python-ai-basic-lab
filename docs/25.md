## 1. 왜 시계열 전용 Transformer가 필요한가?

표준 Transformer의 금융 시계열 한계:
- NLP용으로 설계 → 토큰 단위 처리, 금융의 연속 패턴에 비효율
- 긴 시퀀스(수천 일)에서 Self-Attention의 O(T²) 연산 비용
- 다양한 주기(일별·주별·월별) 패턴을 단일 시퀀스로 처리

### 주요 시계열 Transformer 모델

| 모델 | 특징 | 발표 연도 |
|------|------|---------|
| **Informer** | ProbSparse Attention으로 O(T log T) 달성 | 2021 |
| **Autoformer** | Auto-Correlation으로 주기성 직접 모델링 | 2021 |
| **PatchTST** | 시계열을 패치로 분할 → NLP의 ViT 방식 적용 | 2023 |
| **TFT** | 변수 선택 + 시간적 주의 + 해석 가능성 | 2020 |
| **iTransformer** | 변수 간 상관관계를 Attention으로 모델링 | 2024 |

## 2. PatchTST: 패치 기반 Transformer

ViT(Vision Transformer)의 아이디어를 시계열에 적용:
- 시계열을 **고정 크기 패치(patch)**로 분할
- 패치 단위로 임베딩 → Transformer 입력
- 장점: 시퀀스 길이 감소, 지역적 패턴 포착, 효율적 학습

```
시계열 [T=96]: ─────────────────────────────────
패치(P=16):    [패치1][패치2][패치3][패치4][패치5][패치6]
Transformer:       ↓      ↓      ↓      ↓      ↓      ↓
                 Attention across 6 patches (vs 96 time steps)
```

```python
import torch
import torch.nn as nn
import numpy as np
import pandas as pd
from torch.utils.data import DataLoader, TensorDataset
from sklearn.preprocessing import StandardScaler

class PatchEmbedding(nn.Module):
    """시계열을 패치로 분할하고 임베딩"""
    def __init__(self, patch_size: int, d_model: int, n_features: int):
        super().__init__()
        self.patch_size = patch_size
        self.proj = nn.Linear(patch_size * n_features, d_model)

    def forward(self, x):
        # x: (batch, seq_len, n_features)
        B, T, C = x.shape
        n_patches = T // self.patch_size
        x = x[:, :n_patches * self.patch_size, :]  # 트리밍
        x = x.reshape(B, n_patches, self.patch_size * C)  # (B, n_patches, P*C)
        return self.proj(x)  # (B, n_patches, d_model)


class PatchTST(nn.Module):
    """
    PatchTST: 패치 기반 시계열 Transformer
    - 분류 태스크 (binary/multi-class)
    """
    def __init__(self, n_features: int, seq_len: int, patch_size: int = 8,
                 d_model: int = 64, nhead: int = 4, num_layers: int = 3,
                 n_classes: int = 2, dropout: float = 0.1):
        super().__init__()
        self.patch_embed = PatchEmbedding(patch_size, d_model, n_features)
        n_patches = seq_len // patch_size

        self.cls_token = nn.Parameter(torch.randn(1, 1, d_model))
        self.pos_embed = nn.Parameter(torch.randn(1, n_patches + 1, d_model))
        self.dropout   = nn.Dropout(dropout)

        enc_layer = nn.TransformerEncoderLayer(
            d_model=d_model, nhead=nhead, dim_feedforward=d_model * 4,
            dropout=dropout, batch_first=True, norm_first=True,
        )
        self.encoder = nn.TransformerEncoder(enc_layer, num_layers=num_layers)
        self.head = nn.Sequential(
            nn.LayerNorm(d_model),
            nn.Linear(d_model, n_classes),
        )

    def forward(self, x):
        B = x.size(0)
        x = self.patch_embed(x)                           # (B, n_patches, d_model)
        cls = self.cls_token.expand(B, -1, -1)            # (B, 1, d_model)
        x = torch.cat([cls, x], dim=1)                    # (B, n_patches+1, d_model)
        x = self.dropout(x + self.pos_embed)
        x = self.encoder(x)
        return self.head(x[:, 0, :])                      # CLS 토큰으로 분류
```

## 3. TFT (Temporal Fusion Transformer)

금융 시계열 예측에 특화된 모델로 세 가지 입력 유형을 명시적으로 처리합니다.

| 입력 유형 | 예시 | 처리 방식 |
|---------|------|---------|
| 과거 관측값 | 수익률, 거래량 | LSTM + Attention |
| 미래 알려진 값 | 휴일 여부, 경제 이벤트 날짜 | 직접 입력 |
| 정적 특성 | 섹터, 국가 | 임베딩 후 결합 |

```python
class VariableSelectionNetwork(nn.Module):
    """TFT의 핵심: 입력 특성 중 중요한 것만 선택"""
    def __init__(self, n_vars: int, d_model: int):
        super().__init__()
        self.var_encoders = nn.ModuleList([
            nn.Sequential(nn.Linear(1, d_model), nn.ReLU())
            for _ in range(n_vars)
        ])
        self.gate = nn.Sequential(
            nn.Linear(n_vars * d_model, n_vars),
            nn.Softmax(dim=-1),
        )

    def forward(self, x):
        # x: (batch, n_vars)
        encoded = [enc(x[:, i:i+1]) for i, enc in enumerate(self.var_encoders)]
        combined = torch.cat(encoded, dim=-1)          # (B, n_vars * d_model)
        weights = self.gate(combined)                  # (B, n_vars)
        selected = torch.stack(encoded, dim=1)         # (B, n_vars, d_model)
        output = (weights.unsqueeze(-1) * selected).sum(dim=1)  # (B, d_model)
        return output, weights


# 변수 선택 네트워크 테스트
vsn = VariableSelectionNetwork(n_vars=5, d_model=32)
x_test = torch.randn(8, 5)
out, var_weights = vsn(x_test)
print(f"선택된 변수 가중치: {var_weights[0].detach().numpy().round(3)}")
```

## 4. 실습: PatchTST 주가 방향 예측

```python
# 데이터 준비 (Day 038과 동일한 파이프라인)
np.random.seed(0)
torch.manual_seed(0)

n_days = 2000
price = 100 * np.exp((np.random.randn(n_days) * 0.012).cumsum())
s = pd.Series(price)
ret = s.pct_change().fillna(0)

feat_df = pd.DataFrame({
    'ret_1':  ret,
    'ret_5':  ret.rolling(5).sum(),
    'ret_20': ret.rolling(20).sum(),
    'vol_20': ret.rolling(20).std(),
    'rsi':    ret.clip(lower=0).rolling(14).mean() / ret.abs().rolling(14).mean(),
}).fillna(0)
label = (ret.shift(-1) > 0).astype(int).values

SEQ_LEN   = 64   # PatchTST는 더 긴 시퀀스에서 효과적
PATCH_SIZE = 8   # 8일 패치 → 8개 패치

X_list, y_list = [], []
for i in range(len(feat_df) - SEQ_LEN):
    X_list.append(feat_df.iloc[i:i+SEQ_LEN].values)
    y_list.append(label[i+SEQ_LEN])
X_arr = np.array(X_list, dtype=np.float32)
y_arr = np.array(y_list, dtype=np.int64)

sc = StandardScaler()
X_arr = sc.fit_transform(X_arr.reshape(-1, X_arr.shape[-1])).reshape(X_arr.shape).astype(np.float32)

spl_tr = int(len(X_arr) * 0.7)
spl_va = int(len(X_arr) * 0.85)

X_tr = torch.tensor(X_arr[:spl_tr]); y_tr = torch.tensor(y_arr[:spl_tr])
X_va = torch.tensor(X_arr[spl_tr:spl_va]); y_va = torch.tensor(y_arr[spl_tr:spl_va])
X_te = torch.tensor(X_arr[spl_va:]); y_te = torch.tensor(y_arr[spl_va:])

tl = DataLoader(TensorDataset(X_tr, y_tr), batch_size=64, shuffle=True)
vl = DataLoader(TensorDataset(X_va, y_va), batch_size=256)

# 모델 초기화
patchtst = PatchTST(n_features=5, seq_len=SEQ_LEN, patch_size=PATCH_SIZE,
                     d_model=64, nhead=4, num_layers=3, n_classes=2)
print(f"PatchTST 파라미터: {sum(p.numel() for p in patchtst.parameters()):,}")

# 훈련
opt = torch.optim.AdamW(patchtst.parameters(), lr=5e-4, weight_decay=1e-2)
sch = torch.optim.lr_scheduler.CosineAnnealingLR(opt, T_max=60)
crit = nn.CrossEntropyLoss()
from sklearn.metrics import roc_auc_score

best_auc, best_w = 0, None
for epoch in range(1, 61):
    patchtst.train()
    for X_b, y_b in tl:
        loss = crit(patchtst(X_b), y_b)
        opt.zero_grad(); loss.backward()
        nn.utils.clip_grad_norm_(patchtst.parameters(), 1.0)
        opt.step()
    sch.step()

    patchtst.eval()
    with torch.no_grad():
        va_prob = torch.softmax(patchtst(X_va), dim=1)[:, 1].numpy()
    va_auc = roc_auc_score(y_va.numpy(), va_prob)

    if va_auc > best_auc:
        best_auc = va_auc
        best_w = {k: v.clone() for k, v in patchtst.state_dict().items()}

    if epoch % 15 == 0:
        print(f"Epoch {epoch:3d} | Val AUC={va_auc:.4f}")

patchtst.load_state_dict(best_w)
patchtst.eval()
with torch.no_grad():
    te_prob = torch.softmax(patchtst(X_te), dim=1)[:, 1].numpy()
print(f"\nPatchTST 테스트 AUC: {roc_auc_score(y_te.numpy(), te_prob):.4f}")
```

## 5. 모델 비교: LSTM vs Transformer vs PatchTST

```python
# 간단한 비교 실험 프레임워크
from sklearn.metrics import roc_auc_score

def benchmark_models(models_dict: dict, X_tr, y_tr, X_te, y_te, epochs=30):
    """모델들을 동일한 조건에서 훈련 후 AUC 비교"""
    results = {}
    crit = nn.BCEWithLogitsLoss()

    for name, model in models_dict.items():
        opt = torch.optim.AdamW(model.parameters(), lr=1e-3, weight_decay=1e-2)
        loader = DataLoader(TensorDataset(X_tr, y_tr.float()), batch_size=64, shuffle=True)

        model.train()
        for _ in range(epochs):
            for X_b, y_b in loader:
                out = model(X_b)
                if out.dim() > 1:  # 다중 분류인 경우
                    out = out[:, 1] - out[:, 0]
                loss = crit(out, y_b)
                opt.zero_grad(); loss.backward()
                nn.utils.clip_grad_norm_(model.parameters(), 1.0)
                opt.step()

        model.eval()
        with torch.no_grad():
            logits = model(X_te)
            if logits.dim() > 1:
                prob = torch.softmax(logits, dim=1)[:, 1].numpy()
            else:
                prob = torch.sigmoid(logits).numpy()
        results[name] = roc_auc_score(y_te.float().numpy(), prob)
        print(f"  {name}: AUC={results[name]:.4f}")

    return results
```

## 6. 금융 시계열에서 Transformer 활용 실전 팁

```python
# 팁 1: 앙상블 (다양한 아키텍처 결합)
def ensemble_predict(models: list, X: torch.Tensor) -> np.ndarray:
    all_probs = []
    for model in models:
        model.eval()
        with torch.no_grad():
            out = model(X)
            if out.dim() > 1:
                prob = torch.softmax(out, dim=1)[:, 1]
            else:
                prob = torch.sigmoid(out)
            all_probs.append(prob.numpy())
    return np.mean(all_probs, axis=0)

# 팁 2: 다변량 예측 (여러 종목 동시 예측)
# iTransformer처럼 변수(종목) 축에 Attention 적용
class iTransformerBlock(nn.Module):
    """변수 간 상관관계를 Attention으로 모델링"""
    def __init__(self, n_vars: int, d_model: int, nhead: int):
        super().__init__()
        self.norm   = nn.LayerNorm(d_model)
        self.attn   = nn.MultiheadAttention(d_model, nhead, batch_first=True)
        self.ff     = nn.Sequential(
            nn.Linear(d_model, d_model * 4), nn.GELU(), nn.Linear(d_model * 4, d_model)
        )
        self.norm2  = nn.LayerNorm(d_model)

    def forward(self, x):
        # x: (batch, n_vars, d_model) — 변수 축에 Attention
        residual = x
        x = self.norm(x)
        x, _ = self.attn(x, x, x)
        x = residual + x
        return x + self.ff(self.norm2(x))
```

## 핵심 요약

| 모델 | 시퀀스 길이 | 핵심 아이디어 | 퀀트 적합 상황 |
|------|-----------|------------|-------------|
| Transformer | 짧음~중간 | Self-Attention | 분류 신호 생성 |
| **PatchTST** | 중간~긴 | 패치 분할 | **장기 패턴 학습 권장** |
| Informer | 매우 긴 | Sparse Attention | 초장기 예측 |
| **TFT** | 중간 | 변수 선택 + 해석 | **설명 가능한 예측 필요 시** |
| iTransformer | 중간 | 변수 간 Attention | 다변량 포트폴리오 신호 |
