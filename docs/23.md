## 1. RNN: 순환 신경망

CNN이 공간적 패턴을 학습한다면, RNN은 **시간 의존성(temporal dependency)**을 학습합니다.

$$\mathbf{h}_t = \tanh(\mathbf{W}_h \mathbf{h}_{t-1} + \mathbf{W}_x \mathbf{x}_t + \mathbf{b})$$

```
시간 →    t=1       t=2       t=3       t=T
입력:     x₁  →    x₂  →    x₃  →    x_T
         ↓         ↓         ↓         ↓
은닉:  [h₁] → [h₂] → [h₃] → ... → [h_T]  ← 최종 출력
```

### 기울기 소실 (Vanishing Gradient) 문제

RNN은 시간 단계가 길어질수록 초기 정보가 점점 희석됩니다:
$$\frac{\partial L}{\partial \mathbf{h}_1} = \prod_{t=2}^{T} \frac{\partial \mathbf{h}_t}{\partial \mathbf{h}_{t-1}}$$

$|\frac{\partial \mathbf{h}_t}{\partial \mathbf{h}_{t-1}}| < 1$ 이면 곱셈이 0으로 수렴 → **장기 의존성 학습 불가**.

## 2. LSTM: 장단기 기억 네트워크

3개의 **게이트(gate)**로 정보 흐름을 제어합니다.

```
     ┌─────────────────────────────────────────────────────┐
     │  C_{t-1} ──────────────────────────────────→ C_t   │  ← 셀 상태 (장기 기억)
     │       ↑ forget     ↑ input       ↑                  │
     │  f_t = σ(·)   i_t = σ(·)   o_t = σ(·)            │
     │  g_t = tanh(·)                                      │
     │  C_t = f_t ⊙ C_{t-1} + i_t ⊙ g_t                │
     │  h_t = o_t ⊙ tanh(C_t)                            │
     └─────────────────────────────────────────────────────┘
```

| 게이트 | 수식 | 역할 |
|--------|------|------|
| Forget | $f_t = \sigma(\mathbf{W}_f[\mathbf{h}_{t-1}, \mathbf{x}_t] + b_f)$ | 이전 기억 중 버릴 것 결정 |
| Input  | $i_t = \sigma(\mathbf{W}_i[\mathbf{h}_{t-1}, \mathbf{x}_t] + b_i)$ | 새 정보 중 저장할 것 결정 |
| Output | $o_t = \sigma(\mathbf{W}_o[\mathbf{h}_{t-1}, \mathbf{x}_t] + b_o)$ | 셀 상태 중 출력할 것 결정 |

**GRU (Gated Recurrent Unit)**: LSTM을 단순화한 버전 (게이트 2개). 짧은 시계열에서 LSTM과 비슷한 성능.

## 3. 실습: LSTM으로 주가 수익률 예측

### 데이터 준비 (슬라이딩 윈도우)

```python
import numpy as np
import pandas as pd
import torch
import torch.nn as nn
from torch.utils.data import DataLoader, TensorDataset
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import mean_squared_error
from scipy.stats import spearmanr
import matplotlib.pyplot as plt

np.random.seed(42)
torch.manual_seed(42)

# 주가 시계열 시뮬레이션
n_days = 1500
ret = np.random.randn(n_days) * 0.01
price = 100 * np.exp(ret.cumsum())

# 기술적 팩터 계산
def compute_features(price: np.ndarray) -> np.ndarray:
    s = pd.Series(price)
    ret_1  = s.pct_change().fillna(0)
    ret_5  = ret_1.rolling(5).sum()
    ret_20 = ret_1.rolling(20).sum()
    vol_20 = ret_1.rolling(20).std()
    rsi_14 = (ret_1.clip(lower=0).rolling(14).mean() /
              ret_1.abs().rolling(14).mean()).fillna(0)
    df = pd.concat([ret_1, ret_5, ret_20, vol_20, rsi_14], axis=1)
    df.columns = ['ret_1', 'ret_5', 'ret_20', 'vol_20', 'rsi_14']
    return df.fillna(0).values

features  = compute_features(price)
next_ret  = pd.Series(price).pct_change().shift(-1).fillna(0).values  # 레이블

SEQ_LEN = 30  # 30일 이력 → 다음날 예측

def create_sequences(X, y, seq_len):
    Xs, ys = [], []
    for i in range(len(X) - seq_len):
        Xs.append(X[i:i+seq_len])
        ys.append(y[i+seq_len])
    return np.array(Xs, dtype=np.float32), np.array(ys, dtype=np.float32)

scaler    = StandardScaler()
feat_sc   = scaler.fit_transform(features)
X_seq, y_seq = create_sequences(feat_sc, next_ret, SEQ_LEN)
print(f"시퀀스 형태: {X_seq.shape}")  # (N, 30, 5)

split_tr  = int(len(X_seq) * 0.7)
split_val = int(len(X_seq) * 0.85)

def make_loader(X, y, shuffle=False):
    ds = TensorDataset(torch.tensor(X), torch.tensor(y))
    return DataLoader(ds, batch_size=64, shuffle=shuffle)

train_loader = make_loader(X_seq[:split_tr], y_seq[:split_tr], shuffle=True)
val_loader   = make_loader(X_seq[split_tr:split_val], y_seq[split_tr:split_val])
X_te_t = torch.tensor(X_seq[split_val:])
y_te_np = y_seq[split_val:]
```

### LSTM & GRU 모델

```python
class StockLSTM(nn.Module):
    def __init__(self, input_size: int, hidden_size: int = 64, num_layers: int = 2, dropout: float = 0.2):
        super().__init__()
        self.lstm = nn.LSTM(input_size, hidden_size, num_layers,
                            batch_first=True,
                            dropout=dropout if num_layers > 1 else 0)
        self.dropout = nn.Dropout(dropout)
        self.fc = nn.Sequential(
            nn.Linear(hidden_size, 32), nn.ReLU(), nn.Linear(32, 1),
        )

    def forward(self, x):
        out, _ = self.lstm(x)
        return self.fc(self.dropout(out[:, -1, :])).squeeze(-1)


class StockGRU(nn.Module):
    def __init__(self, input_size: int, hidden_size: int = 64, num_layers: int = 2):
        super().__init__()
        self.gru = nn.GRU(input_size, hidden_size, num_layers, batch_first=True, dropout=0.2)
        self.fc  = nn.Linear(hidden_size, 1)

    def forward(self, x):
        out, _ = self.gru(x)
        return self.fc(out[:, -1, :]).squeeze(-1)
```

### 훈련 및 평가

```python
def train_model(model, n_epochs=50):
    optimizer = torch.optim.Adam(model.parameters(), lr=1e-3)
    scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(optimizer, patience=7, factor=0.5)
    criterion = nn.MSELoss()
    best_mse, best_w = float('inf'), None

    for epoch in range(1, n_epochs + 1):
        model.train()
        for X_b, y_b in train_loader:
            loss = criterion(model(X_b), y_b)
            optimizer.zero_grad(); loss.backward()
            nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            optimizer.step()

        model.eval()
        with torch.no_grad():
            vp = [model(X_b).numpy() for X_b, _ in val_loader]
            vt = [y_b.numpy() for _, y_b in val_loader]
            val_mse = mean_squared_error(np.concatenate(vt), np.concatenate(vp))
        scheduler.step(val_mse)

        if val_mse < best_mse:
            best_mse = val_mse
            best_w = {k: v.clone() for k, v in model.state_dict().items()}

        if epoch % 10 == 0:
            print(f"  Epoch {epoch:3d} | Val MSE={val_mse:.6f}")

    model.load_state_dict(best_w)
    return model

print("LSTM 훈련:")
lstm_model = train_model(StockLSTM(input_size=5))
print("\nGRU 훈련:")
gru_model  = train_model(StockGRU(input_size=5))

# 평가
def evaluate(model, name):
    model.eval()
    with torch.no_grad():
        preds = model(X_te_t).numpy()
    ic     = np.corrcoef(y_te_np, preds)[0, 1]
    ric, _ = spearmanr(y_te_np, preds)
    print(f"{name}: IC={ic:.4f}, Rank IC={ric:.4f}")

evaluate(lstm_model, "LSTM")
evaluate(gru_model,  "GRU")
```

## 4. Attention + LSTM

특정 시점에 더 집중하는 Attention을 LSTM 위에 추가합니다.

```python
class AttentionLSTM(nn.Module):
    def __init__(self, input_size: int, hidden_size: int = 64):
        super().__init__()
        self.lstm      = nn.LSTM(input_size, hidden_size, batch_first=True)
        self.attention = nn.Linear(hidden_size, 1)
        self.fc        = nn.Linear(hidden_size, 1)

    def forward(self, x):
        lstm_out, _  = self.lstm(x)                               # (B, T, H)
        attn_weights = torch.softmax(self.attention(lstm_out), dim=1)  # (B, T, 1)
        context      = (attn_weights * lstm_out).sum(dim=1)       # (B, H)
        return self.fc(context).squeeze(-1), attn_weights.squeeze(-1)

attn_model = AttentionLSTM(input_size=5)
with torch.no_grad():
    pred, weights = attn_model(X_te_t[:4])
print(f"Attention 가중치: {weights.shape}")  # (4, 30)
most_important = weights[0].argmax().item()
print(f"첫 샘플: {SEQ_LEN - most_important}일 전 시점이 가장 중요")
```

## 핵심 요약

| 모델 | 장기 의존성 | 파라미터 | 속도 | 퀀트 적합 상황 |
|------|-----------|---------|------|-------------|
| RNN | 약함 | 적음 | 빠름 | 짧은 시계열 (<10) |
| LSTM | 강함 | 많음 | 보통 | 장기 패턴 학습 |
| GRU | 보통 | 중간 | 빠름 | 실용적 기본값 |
| Attn-LSTM | 강함 | 많음 | 느림 | 해석 가능성 필요 시 |

- **IC > 0.03**: 실전 사용 가능 기준
- **Bidirectional 금지**: 미래 데이터 누설 위험
- **Gradient Clipping**: `clip_grad_norm_(1.0)` 필수
