## 1. 데이터 준비 (FinanceDataReader)

```python
import numpy as np
import pandas as pd
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import accuracy_score
import matplotlib.pyplot as plt

# 삼성전자 실제 데이터 수집
try:
    import FinanceDataReader as fdr
    raw = fdr.DataReader('005930', '2020-01-01', '2024-12-31')
    df  = raw[['Close', 'Volume']].rename(columns={'Close': 'close', 'Volume': 'volume'})
    print(f"✅ 삼성전자: {len(df)}일")
except Exception:
    np.random.seed(42)
    n = 1200
    dates = pd.date_range('2020-01-01', periods=n, freq='B')
    df = pd.DataFrame({
        'close':  np.clip(55000 + np.cumsum(np.random.randn(n) * 800), 40000, 90000),
        'volume': np.random.randint(8_000_000, 25_000_000, n),
    }, index=dates)
    print("⚠️  오프라인 시뮬레이션")

df['ret']        = df['close'].pct_change()
df['log_ret']    = np.log(df['close'] / df['close'].shift(1))
df['ma5']        = df['close'].rolling(5).mean()
df['ma20']       = df['close'].rolling(20).mean()
df['vol_ratio']  = df['volume'] / df['volume'].rolling(10).mean()
df['volatility'] = df['ret'].rolling(5).std()
df = df.dropna()

FEATURE_COLS = ['ret', 'log_ret', 'vol_ratio', 'volatility']
feat = df[FEATURE_COLS].values
n_total = len(feat)
print(f"특성 데이터: {feat.shape}")
```

## 2. Transformer 핵심 블록 구현 (NumPy)

```python
def softmax(x):
    e = np.exp(x - x.max(axis=-1, keepdims=True))
    return e / e.sum(axis=-1, keepdims=True)

def scaled_dot_product_attention(Q, K, V):
    d_k = Q.shape[-1]
    scores  = (Q @ K.T) / np.sqrt(d_k)
    weights = softmax(scores)
    return weights @ V, weights

def multi_head_attention(seq, n_heads=2, d_model=8):
    seq_len, n_feat = seq.shape
    np.random.seed(99)
    head_dim = d_model // n_heads
    all_contexts, all_weights = [], []
    for h in range(n_heads):
        Wq = np.random.randn(n_feat, head_dim) * 0.1
        Wk = np.random.randn(n_feat, head_dim) * 0.1
        Wv = np.random.randn(n_feat, head_dim) * 0.1
        ctx, w = scaled_dot_product_attention(seq @ Wq, seq @ Wk, seq @ Wv)
        all_contexts.append(ctx)
        all_weights.append(w)
    return np.concatenate(all_contexts, axis=-1), all_weights

def positional_encoding(seq_len, d_model):
    pe = np.zeros((seq_len, d_model))
    pos = np.arange(seq_len).reshape(-1, 1)
    dims = np.arange(0, d_model, 2)
    pe[:, 0::2] = np.sin(pos / (10000 ** (dims / d_model)))
    pe[:, 1::2] = np.cos(pos / (10000 ** (dims / d_model)))
    return pe

def transformer_encode(seq, d_model=8, n_heads=2):
    np.random.seed(42)
    W_emb = np.random.randn(seq.shape[-1], d_model) * 0.1
    embedded = seq @ W_emb + positional_encoding(len(seq), d_model)
    attn_out, attn_weights = multi_head_attention(embedded, n_heads, d_model)
    x = embedded + attn_out
    # Feed-Forward
    W1 = np.random.randn(d_model, 16) * 0.1
    W2 = np.random.randn(16, d_model) * 0.1
    x = x + np.maximum(0, x @ W1) @ W2
    return x[-1], attn_weights   # 마지막 시점 출력
```

## 3. Transformer 특성 생성 + 방향 예측

```python
from sklearn.neural_network import MLPClassifier

SEQ_LEN, D_MODEL = 20, 8
X_list, y_list, attn_history = [], [], []

for i in range(SEQ_LEN, n_total - 1):
    seq = feat[i - SEQ_LEN:i]
    encoded, attn_wts = transformer_encode(seq, d_model=D_MODEL)
    X_list.append(np.concatenate([encoded, [feat[i, 0]]]))
    y_list.append(1 if df['ret'].iloc[i + 1] > 0 else 0)
    attn_history.append(attn_wts[0][-1])

X_arr = np.array(X_list)
y_arr = np.array(y_list)

split = int(len(X_arr) * 0.8)
sc = StandardScaler()
X_tr = sc.fit_transform(X_arr[:split])
X_te = sc.transform(X_arr[split:])

clf = MLPClassifier(hidden_layer_sizes=(64, 32), activation='relu',
                    max_iter=400, random_state=42, early_stopping=True)
clf.fit(X_tr, y_arr[:split])
acc = accuracy_score(y_arr[split:], clf.predict(X_te))
print(f"\n삼성전자 Transformer 방향 예측: {acc:.1%}")
```

## 5. LSTM 스타일 vs Transformer 비교

```python
# 기준 모델 (원본 시퀀스 + tanh)
X_base = np.array([feat[i-SEQ_LEN:i].flatten() for i in range(SEQ_LEN, n_total-1)])
sp = int(len(X_base) * 0.8)
sc2 = StandardScaler()
clf_base = MLPClassifier(hidden_layer_sizes=(64, 32), activation='tanh',
                          max_iter=400, random_state=42, early_stopping=True)
clf_base.fit(sc2.fit_transform(X_base[:sp]), y_arr[:sp])
base_acc = accuracy_score(y_arr[sp:], clf_base.predict(sc2.transform(X_base[sp:])))

print(f"\n=== 삼성전자 방향 예측 비교 ===")
print(f"LSTM 스타일 (tanh+시퀀스): {base_acc:.1%}")
print(f"Transformer (Self-Att):   {acc:.1%}")
print(f"개선 효과: {(acc - base_acc)*100:+.1f}%p")

models = ['LSTM 스타일', 'Transformer']
colors = ['steelblue', '#6366f1']
plt.figure(figsize=(6, 4))
bars = plt.bar(models, [base_acc*100, acc*100], color=colors, alpha=0.85)
for bar, a in zip(bars, [base_acc, acc]):
    plt.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 0.3,
             f'{a:.1%}', ha='center', va='bottom', fontweight='bold')
plt.ylim(45, 70)
plt.ylabel('테스트 정확도 (%)')
plt.title('삼성전자 주가 예측: LSTM vs Transformer')
plt.tight_layout()
plt.savefig('transformer_vs_lstm.png', dpi=120)
```

## 핵심 정리

| 구성 요소 | 역할 |
|-----------|------|
| Positional Encoding | 날짜 순서 정보를 벡터에 추가 |
| Multi-Head Attention | 여러 관점에서 날짜 간 중요도 계산 |
| Feed-Forward | 비선형 패턴 추출 |
| Residual Connection | 깊은 모델에서 기울기 소실 방지 |
